interacting with your vcs (option 1)



            buildscript {
                repositories { jcenter() }
                dependencies {  classpath 'org.ajoberstar:gradle-git:0.8.0' }
            }

            import org.ajoberstar.grgit.*
            ext.repo = Grgit.open(file('.'))

            task tagRelease << {
                repo.tag.add {
                  name = version
                  message = "Release of ${version}"
                }
            }
